52 def self.literal_duration(duration)
53   h = Hash.new(0)
54   duration.parts.each{|unit, value| h[unit] += value}
55   s = String.new
56 
57   DURATION_UNITS.each do |unit|
58     if (v = h[unit]) != 0
59       s << "#{v.is_a?(Integer) ? v : sprintf('%0.6f', v)} #{unit} "
60     end
61   end
62 
63   if s.empty?
64     '0'
65   else
66     s
67   end
68 end
