60 def changed_columns
61   changed = super
62   if vh = @values_hashes
63     values = @values
64     changed = changed.dup if frozen?
65     vh.each do |c, v|
66       match = values.has_key?(c) && v == values[c].hash
67       if changed.include?(c)
68         changed.delete(c) if match
69       else
70         changed << c unless match
71       end
72     end
73   end
74   changed
75 end
