363 def fetch_rows(sql)
364   execute(sql) do |result|
365     cps = db.conversion_procs
366     type_procs = result.types.map{|t| cps[base_type_name(t)]}
367     j = -1
368     cols = result.columns.map{|c| [output_identifier(c), type_procs[(j+=1)]]}
369     self.columns = cols.map(&:first)
370     max = cols.length
371     result.each do |values|
372       row = {}
373       i = -1
374       while (i += 1) < max
375         name, type_proc = cols[i]
376         v = values[i]
377         if type_proc && v
378           v = type_proc.call(v)
379         end
380         row[name] = v
381       end
382       yield row
383     end
384   end
385 end
